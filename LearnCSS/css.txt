layout:
	quy tắc vàng:
		các thẻ cha ko nên set width height, để thẻ con xa nhất (trong cùng) set display:block và (padding, margin,border) để dãn thẻ cha
		các thẻ cha được cố định bởi width height thì để thẻ con gần nhất set width height 100% hay 50% để lấp đầy thẻ cha
		thẻ inline  ko thể set width height
		
	bước 1: tạo bố cục layout: header, nav, content-column1, content-column2, content-column3, footer
		=> sử dụng background; padding cho từng phần để nhìn bố cục
		=> sử dụng (flex, height) lên thẻ main; thẻ con flex:1;
	bước 2: responsive cho bố cục layout:
		sử dụng (flex-direction: column) cho thẻ main
	bước 3: tạo phần tử:
		header nav:
			dựng thẻ ul li a
			xây dựng (overflow; float)
			thẻ a dùng (block; padding)
		main nav-left:
			thẻ a dùng (block; padding)
	bước 4: responsive cho phần tử
		header nav: 	sử dụng float: none; để tạo nav column
		main nav-left:	sử dụng display:none để ẩn và hiện toggle
		main nav-cvghfright
	bước 3: tạo tiếp khung cho các element con trong khung thẻ div
	bước 4: responsive
		ẩn nav left ở 1024px hiện nút toggle; 
	bước 5: xây dựng chi tiết các element
	
	flex:	ko nên chuyển đổi row thành column hay ngược lại trong quá trình reponse để tránh làm phức tạp; thay vào đó sử dụng grow; width;...
		sử dụng line-height thay height cho thẻ chứa text; đảm bảo có thể center text
		flex: 0 0 200px"	=> grow shrink basis
		flex-flow: row wrap;	=> cho xắp xếp theo trục x và cho phép xuống dòng
		xếp theo trục x:	flex-direction: row;		=> row-reverse (đảo ngược thứ tự item)
		xếp theo trục y:	flex-direction: column;		=> column-reverse (đảo ngược thứ tự item)
		cho phép xuống dòng:	flex-wrap: wrap;		=> mặc định nowrap (không cho xuống dòng) => wrap-reverse theo thứ tự ngược lại
			- nếu không cho xuống dòng thì item sẽ bị nén kích thước content lại
		căn chỉnh giữa các item trục x: justify-content: center	=> đẩy item vào giữa; flex-start(đẩy vào đầu); flex-end(đẩy vào cuối); space-around (đẩy đều)
									space-between (đẩy đều nhưng ko tạo khoảng trống 2 đầu); 
		căn chỉnh giữa các item trục y:  align-items: flex-end	=> flex-start; flex-end; flex-center; flex-stretch (mặc định)
		căn chỉnh giữa các row: align-content: space-between;	=> space-around; stretch; center; flex-start; flex-end
		xắp xếp theo thứ tự:
			sử dụng trên thẻ item:
				order: index		=> xắp xếp theo index
				flex-grow: 1		=> chia khoảng dư thừa trên row cho các item theo tỉ lệ
				flex-shrink: 1		=> tỉ lệ item sẽ co lại so với phần còn lại, nếu = 0 thì sẽ ko co, nếu =1 thì tỉ lệ co = item khác
				flex-base:100px		=> chỉ định kích thước content
				align-self: center	=> tự căn chỉnh 
	min-width:200px; max-width: 500px;		=> thiết lập độ co dãn theo chiều ngang, và gắn width là max-width nếu ko set width
							=> nếu chỉ có min-width:200px; thì width có thể co từ 200px -> + vô cùng
							=> nếu chỉ có max-width:500px; thì width có thể co từ 0 -> 500px
	position:					=> sử dụng thuộc tính left, top, right, bottom để định vị
		relative;				=> định vị theo vị trí hiện tại của nó, các thuộc tính khác ko bị điều chỉnh
		fixed;					=> định vị theo cửa sổ trình duyệt; ko để lại khoảng trống trên trang
		absolute;				=> định vị theo thẻ cha(thẻ cha phải là relative); nếu ko có thì sẽ định vị theo thẻ body
		sticky;					=> định vị theo khung thanh cuộn, chỉ có tác dụng khi có thanh cuộn
		static; 				=> luôn được định vị bằng dòng bình thường của trang, ko bị ảnh hưởng bởi top, bottom, left, and right
		z-index: 1;				=> nổi lên màn hình; số càng lớn nổi càng cao; chỉ có tác dụng khi có position
	Căn chỉnh phần tử theo margin:auto
		bước 1: biến phần tử thành block
		bước 2: set width của phần tử nhỏ hơn width của thẻ cha
		bước 3: margin-left: auto		=> đẩy phần tử về phía phải
			margin-rigth: auto		=> đẩy về phí trái
			margin-left và right:auto	=> đẩy vào giữa
	overflow:					=> xử lý nội dung bị tràn ra ngoài thẻ cha
		scroll					=> thêm thanh cuộn cho nội dung bị tràn
		hidden					=> phần tràn bị cắt
		auto					=> thêm thanh cuộn giống scroll; nhưng chỉ thêm khi cần thiết
		overflow-x: scroll;			=> xử lý tràn theo trục x
  		overflow-y: scroll;			=> xử lý tràn theo trục y
	visibility: hidden;				=> ẩn thẻ, nhưng vẫn để lại 1 khoảng trắng của phần tử đó
	display: inline;				=> inline-block; block
		inline:				
			không thể set width height
			top và bottom của padding/margin không tác động tới element khác
			không có dấu ngắt dòng
		inline-block:
			có thể set width height
			không có dấu ngắt dòng
		block:
			có thể set width height
			có dấu ngắt dòng
		none:
			ẩn phần tử, như phần tử đó chưa tồn tại
	float:
		giống game xếp gạch
		các thẻ có float sẽ trở thành inline-block
		thẻ không có float sẽ đè lên thẻ có float
			sử dụng clear: left; trên thẻ ko có float để không đè
		thẻ cha(ko phải float) chứa thẻ float không thể bao bọc thẻ hết thẻ float nếu nó quá lớn
			để khắc phục điều này:	sử dụng trên thẻ cha: overflow: auto;
	sử dụng trên thẻ cha:
		float: right					=> các phần tử con đều có thuộc tính float: right
